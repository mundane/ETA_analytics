        function generateData7() {
			"use strict";
            var today = new Date(),
                before = today.add(Date.DAY, -200),
                data = [{
                    date: before,
                    num: 0,
                    djia: 10000,
                    sp500: 1100
                }],
                i, currentDate = before;

            for (i = 1; i < 200; i++) {
                data.push({
                    date: (currentDate = currentDate.add(Date.DAY, 1)),

                    sp500: data[i - 1].sp500 + ((Math.floor(Math.random() * 2) % 2) ? -1 : 1) * Math.floor(Math.random() * 7),
